1- function timeoutPromiseResolve(interval) {
         return new Promise((resolve, reject) => {
            setTimeout(function () {
              resolve("successful");
             }, interval);
         });
      }
              
2- function timeoutPromiseReject(interval) {
          return new Promise((resolve, reject) => {
            setTimeout(function () {
              reject("error");
             }, interval);
         });
     }
              
 3- function timeTest() {
        timeoutPromiseResolve(5000);
        timeoutPromiseReject(2000);
        timeoutPromiseResolve(3000);
